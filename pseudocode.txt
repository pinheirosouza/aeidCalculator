            var start <- tempo nesse momento

            var cont_a, cont_b, somaproximo, lista_a, lista_b, tam, resultado, intResultado

            para cada numero em a como String:
                lista_a[último] <- numero como Inteiro
            
            para cada numero em b como String:

            cont_a <- tamanho lista_a
            cont_b <- tamanho lista_b

            se cont_a igual cont_b:
                passa
            
            se cont_a menor que cont_b:
                para cada n em variação(cont_b - cont_a):
                    lista_a[início] <- 0 
            
            se cont_a maior que cont_b:
                para cada n em variação(cont_a - cont_b):
                    lista_b[início] <- 0
            
            cont_a = tamanho da lista_a - 1
            cont_b = tamanho da lista_b - 1
            tam = tamanho da lista_a - 1

            para cada chave e valor em lista_a transformada em Dict com chaves igual index:
                se lista_a[cont_a] + lista_b[cont_b] + somaproximo menor que 10:
                    lista_a[início] <- lista_a[cont_a] + lista_b[cont_b] + somaproximo
                    somaproximo <- 0

                caso lista_a[cont_a] + lista_b[cont_b] + somaproximo igual ou maior que 10:
                    se chave igual tam:
                        lista_a[início] <- módulo da divisão ((lista_a[cont_a] + lista_b[cont_b] + somaproximo)/10)
                        lista_a[início] <- 1
                    senão:
                        lista_a[início] <- módulo da divisão ((lista_a[cont_a] + lista_b[cont_b] + somaproximo)/10)
                        somaproximo <- 1

                cont_a <- cont_a -1
                cont_b <- cont_b -1

                var res = união dos algarismos em uma só String transformando para inteiro
                retorna res
            }
            var end <- tempo nesse momento

            retorna intResultado, (end-start)
        }
